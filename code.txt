        if (!tableName) {
            c.fields = [];
            return;
        }
        var ga = new GlideAjax('global.getChartRevamp');
        ga.addParam('sysparm_name', 'getFields');
        ga.addParam('sysparm_tableid', tableName);
        ga.getXMLAnswer(function(response) {
            var answer = JSON.parse(response);
            c.fields = answer || [];
            $scope.$apply();
        });
        c.fetchFields();
    };
